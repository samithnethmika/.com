#define enA 6
#define in1 8
#define in2 7
#define in3 4
#define in4 3
#define enB 5


#define servo1 11
#define servo2 10
#define servo3 9

Servo mservo1, mservo2, mservo3;
int pos = 0;
char Incoming_value = 0;
char usechar = "0";

void setup() {
Serial.begin(9600);
// put your setup code here, to run once:

pinMode(enA, OUTPUT);
pinMode(in1, OUTPUT);
pinMode(in2, OUTPUT);
pinMode(in3, OUTPUT);
pinMode(in4, OUTPUT);
pinMode(enB, OUTPUT);


digitalWrite(in1, LOW);
digitalWrite(in2, HIGH);
digitalWrite(in3, LOW);
digitalWrite(in4, HIGH);

mservo1.attach(servo1); // 90-180
mservo2.attach(servo2); // 10-90
mservo3.attach(servo3); // 70-170

mservo1.write(90);
mservo2.write(40);
mservo3.write(70);


}
int srv1 = 90;
int srv2 = 40;
int srv3 = 70;
void loop()
{

if(Serial.available() > 0)
{
Incoming_value = Serial.read();
//Serial.println(Incoming_value);
usechar = Incoming_value;

}

if(usechar=='0')
{
analogWrite(enA, 0);
digitalWrite(in1, HIGH);
digitalWrite(in2, LOW);

analogWrite(enB, 0);
digitalWrite(in4, HIGH);
digitalWrite(in3, LOW);

}
if(usechar=='A')
{
analogWrite(enA, 120);
digitalWrite(in1, HIGH);
digitalWrite(in2, LOW);

analogWrite(enB, 120);
digitalWrite(in4, HIGH);
digitalWrite(in3, LOW);
}

if(usechar=='B')
{
analogWrite(enA, 120);
digitalWrite(in1, LOW);
digitalWrite(in2, HIGH);

analogWrite(enB, 120);
digitalWrite(in4, LOW);
digitalWrite(in3, HIGH);
}

if(usechar=='C')
{
analogWrite(enA, 120);
digitalWrite(in1, LOW);
digitalWrite(in2, HIGH);

analogWrite(enB, 120);
digitalWrite(in4, HIGH);
digitalWrite(in3, LOW);
}

if(usechar=='D')
{
analogWrite(enA, 120);
digitalWrite(in1, HIGH);
digitalWrite(in2, LOW);

analogWrite(enB, 120);
digitalWrite(in4, LOW);
digitalWrite(in3, HIGH);
}

if(usechar=='G') {
if(srv1<=180)
{
srv1++;
mservo1.write(srv1);

delay(10);
}
}

if(usechar=='J')
{
if(srv1>=90)
{
srv1--;
mservo1.write(srv1);

delay(10);
}
}

//-----
if(usechar=='F')
{
if(srv2<=120)
{
srv2++;
mservo2.write(srv2);

delay(10);
}
}

if(usechar=='I')
{
if(srv2>=0)
{
srv2--;
mservo2.write(srv2);


delay(10);
}
}

//----
if(usechar=='E')
{
if(srv3<=160)
{
srv3++;
mservo3.write(srv3);

delay(10);
}
}
if(usechar=='H')
{
if(srv3>=80) {
srv3--;
mservo3.write(srv3);

delay(10);
}
}

}
